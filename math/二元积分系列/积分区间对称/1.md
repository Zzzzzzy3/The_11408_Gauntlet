$$\begin{aligned}
& \text{由积分区域对称性可得} \\
& \quad I=\frac{1}{2}[\iint_D(\frac{(x-2y)^2}{\sqrt{x^2+y^2}}-\frac{(y-2x)^2}{\sqrt{x^2+y^2}})] \\\\
& \text{转化为极坐标},D_1\text{由y=x对称得二重积分为}\frac{1}{2}I \\
& \quad D=\{\rho\leq ({ sin\theta+ cos\theta})^{\frac{1}{3}}\},~D_1\{\rho\leq ({ sin\theta+ cos\theta})^{\frac{1}{3}},~-\frac{\pi}{4}\leq \theta \leq \frac{\pi}{4}\}\\
& \text{解得} \\
& \quad I=\frac{11\sqrt{2}}{9}
\end{aligned}
